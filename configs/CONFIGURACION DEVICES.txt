Configuración R1-Cisco7200

#configuración de interfaces
enable
configure terminal
interface f0/0
ip address 10.10.0.1 255.255.255.0
no shutdown
do write
exit

interface g1/0
ip address 10.10.1.1 255.255.255.252
no shutdown
do write
exit

interface g2/0
ip address 10.10.2.1 255.255.255.252
no shutdown
do write
exit

#Configuración de enrutamiento 
router eigrp 100
network 10.10.0.0 0.0.0.255
network 10.10.2.0 0.0.0.3
network 10.10.1.0 0.0.0.3
no auto-summary
exit

router ospf 1
 router-id 1.1.1.1
 network 10.10.0.0 0.0.0.255 area 0
 network 10.10.1.0 0.0.0.3 area 0
 network 10.10.2.0 0.0.0.3 area 0
 redistribute eigrp 100 subnets

exit

do write



#Configuración de acceso a SNMP para monitoreo
access-list 99 permit 10.10.0.0 0.0.0.255
access-list 99 permit 10.10.1.0 0.0.0.3
access-list 99 permit 10.10.2.0 0.0.0.3
snmp-server community EPN ro 99
snmp-server ifindex persist
do write
exit

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
Configuración R2-MikroTikCHR7

#configuracion de interfaces
/ip address add address=10.10.1.2/30 interface=ether1
/ip address add address=192.168.1.1/24 interface=ether2

#configuracion de enrutamiento
/routing ospf instance
add name=v2inst version=2 router-id=10.10.1.2

/routing ospf area
add name=backbone_v2 area-id=0.0.0.0 instance=v2inst

/routing ospf interface-template
add networks=10.10.1.0/30 area=backbone_v2
add networks=192.168.1.0/24 area=backbone_v2

#configuración de SNMP para monitoreo
/snmp set enabled=yes
/snmp community add name=EPN addresses=10.0.0.0/24,10.10.1.0/30,192.168.1.0/24 read-access=yes

Configuración R3-Cisco7200

no shutdown

no shutdown

exit

#Configuración de enrutamiento
router eigrp 100
network 10.10.2.0 0.0.0.3
network 192.168.2.0 0.0.0.255
no auto-summary

exit

 network 10.10.2.0 0.0.0.3 area 0

exit
